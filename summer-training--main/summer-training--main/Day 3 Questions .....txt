3....factorial.
answer :

#include<iostream>
using namespace std;
int main(){
    int n;
    int fac=1;
    cout<<"Enter the number : ";
    cin>>n;
    for(int i=n;i>=1;i--){
        fac=fac*i;
    }
    cout<<"The factorial : "<<fac;

}



4....reverse number.
answer :


#include<iostream>
using namespace std;
int main(){
    int n;
    
    int digit;
    cout<<"Enter the number : ";
    cin>>n;
    
    int rev=0;
    for(int i=n;i>0;){
        digit=n%10;
        rev=rev*10+digit;
        n=n/10;
    }
    cout<<rev;
}

6....find prime number.
answer :

#include<iostream>
using namespace std;
int main(){
    int n;
    cout<<"Enter the number : ";
    cin>>n;
    int count=0;
    for(int i=1;i<=n;i++){
        if(n%i==0){
            count++;
        }
    }
    if(count==2){
        cout<<"It is a prime number";
    }
    else{
        cout<<"It is not a prime number ";
    }

}
