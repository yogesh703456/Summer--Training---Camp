6....simple calculator.
answer :


#include<iostream>
using namespace std;
int main(){
    cout<<"Enter the sign + - / * \n";
    int a,b,c;
    char n;
    
    cout<<"Enter the first number : ";
    
    cin>>a;
    cout<<"Enter the second number : ";
    cin>>b;
    cout<<"Enter the sign : ";
    cin>>n;
    
    if(n=='+'){
        c=a+b;
        cout<<c;
        
    }
    else if(n=='-'){
        c=a-b;
        cout<<c;
    }
    else if(n=='*'){
        c=a*b;
        cout<<c;
    }
    else if(n=='/'){
        c=a/b;
        cout<<c;
    }
    else{
        cout<<"error,Enter the valid sign";
    }
    
}
